def split_string(string)

  count=string.length

  (2..count).step(3) do |x|
    string.insert(x, ',')
  end

  string.split(',')

end

p split_string("abcdef")   # ["ab", "cd", "ef"]
p split_string("abcdefg")  # ["ab", "cd", "ef", "g_"]
p split_string("")         # []

def split_string(string)

  count=string.length
  string = string + "_" if not count % 2 == 0

  j=2

  for i in 2..(count/2+1) do
    string.insert(j, ',')
    j+=3
  end

  string.split(',')

end

p split_string("abcdef")   # ["ab", "cd", "ef"]
p split_string("abcdefg")  # ["ab", "cd", "ef", "g_"]
p split_string("")         # []
