def add_zero(num, amount)
  count=num.to_s.length
  "0" *(amount-count) + num.to_s
end

p add_zero(3, 4)     # "0003"
p add_zero(23, 6)    # "000023"
p add_zero(10, 4)    # "0010"
p add_zero(253, 5)   # "00253"

A1

def add_zero(num, amount)

  result=""
  count=num.to_s.length

  for i in 1..(amount-count) do
    result = result + "0"
  end

  result << num.to_s

end

p add_zero(3, 4)     # "0003"
p add_zero(23, 6)    # "000023"
p add_zero(10, 4)    # "0010"
p add_zero(253, 5)   # "00253"
